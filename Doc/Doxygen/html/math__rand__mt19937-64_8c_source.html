48 #include<stdio.h>

50 #define NN 312
51 #define MM 156
52 #define MATRIX_A 0xB5026F5AA96619E9ULL
53 #define UM 0xFFFFFFFF80000000ULL /* Most significant 33 bits */
54 #define LM 0x7FFFFFFFULL /* Least significant 31 bits */
55 
56 
57 /* The array for the state vector */
58 static unsigned long long mt[NN];
59 /* mti==NN+1 means mt[NN] is not initialized */
60 static int mti=NN+1;
61 
62 /* initializes mt[NN] with a seed */
63 void init_genrand64(unsigned long long seed)
64 {
65  mt[0] = seed;
66  for (mti=1; mti<NN; mti++)
67  mt[mti] = (6364136223846793005ULL * (mt[mti-1] ^ (mt[mti-1] >> 62)) + mti);
68 }
69 
70 /* generates a random number on [0, 2^64-1]-interval */
71 unsigned long long genrand64_int64(void)
72 {
73  int i;
74  unsigned long long x;
75  static unsigned long long mag01[2]={0ULL, MATRIX_A};
76 
77  if (mti >= NN) { /* generate NN words at one time */
78 
79  /* if init_genrand64() has not been called, */
80  /* a default initial seed is used */
81  if (mti == NN+1)
82  init_genrand64(5489ULL);
83 
84  for (i=0;i<NN-MM;i++) {
85  x = (mt[i]&UM)|(mt[i+1]&LM);
86  mt[i] = mt[i+MM] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
87  }
88  for (;i<NN-1;i++) {
89  x = (mt[i]&UM)|(mt[i+1]&LM);
90  mt[i] = mt[i+(MM-NN)] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
91  }
92  x = (mt[NN-1]&UM)|(mt[0]&LM);
93  mt[NN-1] = mt[MM-1] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
94 
95  mti = 0;
96  }
97 
98  x = mt[mti++];
99 
100  x ^= (x >> 29) & 0x5555555555555555ULL;
101  x ^= (x << 17) & 0x71D67FFFEDA60000ULL;
102  x ^= (x << 37) & 0xFFF7EEE000000000ULL;
103  x ^= (x >> 43);
104 
105  return x;
106 }
107 
108 /* generates a random number on [0,1]-real-interval */
109 double genrand64_real1(void)
110 {
111  return (genrand64_int64() >> 11) * (1.0/9007199254740991.0);
112 }
