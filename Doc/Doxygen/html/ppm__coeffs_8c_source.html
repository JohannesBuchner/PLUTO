332  int indx[16];
333  double rp, rc, rm, vol, d, a[16];
334  static double **beta;
335 
336 /* ----------------------------------------------------
337  set the reconstruction stencil & order of accuracy
338  ---------------------------------------------------- */
339 
340  #if PPM_ORDER == 3
341  iL = 1; iR = 1;
342  #elif PPM_ORDER == 4
343  iL = 1; iR = 2;
344  #elif PPM_ORDER == 5
345  iL = 2; iR = 2;
346  #endif
347  n = iR + iL + 1;
348 
349  beg = iL;
350  end = grid->np_tot[dir] - 1 - iR;

355  }
356 
357 /* -------------------------------------------------------
358  grid_type is used to select the Jacobian J
359  of the 1D coordinate direction:
360 
361  grid_type = 0 --> geometry is Cartesian (J = 1)
362  grid_type = 1 --> geometry is cylindrical in the
363  radial coordinate (J = r)
364  grid_type = 2 --> geometry is spherical in the
365  radial coordinate (J = r^2)
366  grid_type = -1 --> geometry is spherical in the
367  meridional coordinate (J = sin(theta))
368  ------------------------------------------------------- */
369 
370  grid_type = 0; /* Default */
371  #if GEOMETRY == CYLINDRICAL || GEOMETRY == POLAR
372  if (dir == IDIR) grid_type = 1;
373  #elif GEOMETRY == SPHERICAL
374  if (dir == IDIR) grid_type = 2;
375  else if (dir == JDIR) grid_type = -1;

377 
378 /* -----------------------------------------------
379  Start main loop
380  ----------------------------------------------- */
381 
382  for (i = beg; i <= end; i++){
383 
384  rc = grid->x[dir][i];
385  jb = i - iL; je = i + iR;
386 
387  for (j = jb; j <= je; j++){ /* -- stencil loop -- */
388  rp = grid->xr[dir][j];
389  rm = grid->xl[dir][j];
390  switch (grid_type) {
391  case 0:
392  vol = (rp - rm);
393  for (k = 0; k < n; k++) { /* -- order loop -- */
394  beta[k][j-jb] = (pow(rp-rc,k+1) - pow(rm-rc,k+1))/(k+1.0)/vol;
395  }
396  break;
397 
398  case 1:
399  vol = (rp*rp - rm*rm)/2.0;
400  for (k = 0; k < n; k++) { /* -- order loop -- */
401  beta[k][j-jb] = pow(rp-rc,k+1)*((k+1.0)*rp + rc)
402  -pow(rm-rc,k+1)*((k+1.0)*rm + rc);
403  beta[k][j-jb] /= (k+2.0)*(k+1.0)*vol;
404  }
405  break;
406 
407  case 2:
408  vol = (rp*rp*rp - rm*rm*rm)/3.0;
409  for (k = 0; k < n; k++) { /* -- order loop -- */
410  beta[k][j-jb] = pow(rp-rc,k+1)*((k*k + 3.0*k + 2.0)*rp*rp
411  + 2.0*rc*(k+1.0)*rp + 2.0*rc*rc)
412  -pow(rm-rc,k+1)*((k*k + 3.0*k + 2.0)*rm*rm
413  + 2.0*rc*(k+1.0)*rm + 2.0*rc*rc);
414 
415  beta[k][j-jb] /= (k+3.0)*(k+2.0)*(k+1.0)*vol;
416  }
417  break;
418 
419  case -1:
420  vol = cos(rm) - cos(rp);

422  for (s_k = 0; s_k < n; s_k++){

426  }
427  beta[0][j-jb] /= vol;
428  beta[1][j-jb] /= vol;
429  beta[2][j-jb] /= vol;
