657  + (B*B - e_P*b2*a*a)*(phi[mu] + 2.0*a*u[mu])*eta_1;
658  }
659  }

667  }

690  Since \phi_a = (-lambda,1), the only non-vanishing
691  components, for a=0,1,2,3 are:
692

695

697  = lambda*(u^2b^3 - u^3b^2)
698

700  + eps^{2103} \phi_1 (u_0b_3 - u_3b_0) =
701  = -lambda*(u_1b_3 - u_3b_1) + B3
702 
703  d^3 = eps^{3012} \phi_0 (u_1b_2 - u_2b_1)
704  + eps^{3102} \phi_1 (u_0b_2 - u_2b_0) =
705  = lambda*(u_1b_2 - u_2b_1) - B2
706 
707  Where
708 
709  \eps^{0123} = \eps^{2013} = \eps{3102} = 1
710  \eps^{1023} = \eps^{2103} = \eps{3012} = -1
711 
712  and B^k = b^ku^0 - u^kb^0 is the lab frame field.
713  ----------------------------------------------------- */
714

716  if (k != KALFVP && k != KALFVM) continue;
717 
718  a = u[0]*(q[VXn] - lambda[k]);

726  + (u[0]*b[3] - u[3]*b[0]);

728  - (u[0]*b[2] - u[2]*b[0]);
729

732  if (degen1 && k == KALFVM) s = -1.0;

735  Ru[5] = Ru[1]*s;

737  Ru[7] = Ru[3]*s;
738  Ru[8] = Ru[9] = 0.0;

741 
742  RR[RHO][k] = 0.0;
743  EXPAND(RR[VXn][k] = (-q[VXn]*Ru[0] + Ru[1])/u[0]; ,
744  RR[VXt][k] = (-q[VXt]*Ru[0] + Ru[2])/u[0]; ,
745  RR[VXb][k] = (-q[VXb]*Ru[0] + Ru[3])/u[0];)
746  RR[PRS][k] = Ru[8];
747 
748  EXPAND( ,
749  RR[BXt][k] = b[2]*Ru[0] - b[0]*Ru[2] - u[2]*Ru[4] + u[0]*Ru[6]; ,
750  RR[BXb][k] = b[3]*Ru[0] - b[0]*Ru[3] - u[3]*Ru[4] + u[0]*Ru[7];)
751  }
752 
753  /* -------------------------
754  ~ Entropy wave ~
755  ------------
756  ------------------------- */
757 
758  k = KENTRP;
759  RR[RHO][k] = 1.0;
760

762  ~ Divergence wave ~
763  ---------------
764  ------------------------- */
765 
766  #ifdef KDIVB
767  #if DIVB_CONTROL != EIGHT_WAVES
768  k = KDIVB;
769  RR[BXn][k] = 1.0;
770  #endif
771  #endif
772

774  Now find the left eigenvectors by performing
775  LU decomposition
776  ---------------------------------------------------- */
777 
778  for (nv = NFLX; nv--; ){
779  for (k = NFLX; k--; ){
780  tmp[nv][k] = RR[nv][k];
781  }}

792  }}

795  LL[KDIVB][BXn] = RR[BXn][KDIVB] = 0.0;
796  #endif
797  #endif
798  return;
799  }
