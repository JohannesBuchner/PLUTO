104 // (or advanceSolution and SWEEP in the next future)

110  double ***dV[CHOMBO_NDV];
111 
112  for (int dir = 0; dir < SpaceDim; ++dir){
113  grid->level = m_level;
114  grid->np_tot[dir] = a_box.size()[dir]+2*m_numGhost;
115  grid->np_int[dir] = a_box.size()[dir];
116  grid->np_tot_glob[dir] = m_domain.domainBox().size()[dir]+2*m_numGhost;
117  grid->np_int_glob[dir] = m_domain.domainBox().size()[dir];
118  grid->nghost[dir] = m_numGhost;
119  grid->beg[dir] = a_box.loVect()[dir]+m_numGhost;
120  grid->end[dir] = a_box.hiVect()[dir]+m_numGhost;
121  grid->gbeg[dir] = m_domain.domainBox().loVect()[dir]+m_numGhost;
122  grid->gend[dir] = m_domain.domainBox().hiVect()[dir]+m_numGhost;
123  grid->lbeg[dir] = m_numGhost;
124  grid->lend[dir] = grid->np_int[dir] + m_numGhost - 1;
125 
126  grid->lbound[dir] = 0;
127  grid->rbound[dir] = 0;
128  // Set flags to compute boundary conditions
129  // is_gbeg (left boundary)
130  Box tmp = a_box;
131  tmp.shift(dir,-1);
132  tmp &= m_domain;
133  tmp.shift(dir,1);
134  if (tmp != a_box) grid->lbound[dir] = 1;
135  // is_gend (right_boundary)
136  tmp = a_box;
137  tmp.shift(dir,1);
138  tmp &= m_domain;
139  tmp.shift(dir,-1);
140  if (tmp != a_box) grid->rbound[dir] = 1;
141 
142  }
143 
144  for (int dir = 0; dir < 3; ++dir){
145  if (dir >= SpaceDim) {
146  grid->np_int[dir] = grid->np_int_glob[dir] = 1;
147  grid->np_tot[dir] = grid->np_tot_glob[dir] = 1;
148  grid->nghost[dir] = 0;
149  grid->beg[dir] = grid->end[dir] = 0;
150  grid->gbeg[dir] = grid->gend[dir] = 0;
151  grid->lbeg[dir] = grid->lend[dir] = 0;
152  grid->lbound[dir] = grid->rbound[dir] = 0;
153  }
154  np_tot_glob = grid->np_tot_glob[dir];
155  np_int_glob = grid->np_int_glob[dir];
156  np_tot = grid->np_tot[dir];
157  np_int = grid->np_int[dir];
158 
159  grid->x[dir] = ARRAY_1D(np_tot, double);
160  grid->xr[dir] = ARRAY_1D(np_tot, double);
161  grid->xl[dir] = ARRAY_1D(np_tot, double);
162  grid->dx[dir] = ARRAY_1D(np_tot, double);
163 
164  stretch = 1.;
165  if (dir == JDIR) stretch = g_x2stretch;
166  if (dir == KDIR) stretch = g_x3stretch;
167 
168  for (i = 0; i < np_tot; i++){
169  #if (CHOMBO_LOGR == YES)
170  if (dir == IDIR) {
171  double argl = Real(i + grid->beg[dir] - 2*grid->nghost[dir]);
172  double argr = Real(i + grid->beg[dir] - 2*grid->nghost[dir] + 1);
173 
174  grid->xl[dir][i] = g_domBeg[IDIR]*exp(argl*m_dx);
175  grid->xr[dir][i] = g_domBeg[IDIR]*exp(argr*m_dx);
176  grid->x[dir][i] = 0.5*(grid->xr[dir][i] + grid->xl[dir][i]);
177  grid->dx[dir][i] = grid->xr[dir][i] - grid->xl[dir][i];
178  } else {
179  #endif
180  double argl = Real(i + grid->beg[dir] - 2*grid->nghost[dir]);
181  grid->xl[dir][i] = argl*m_dx*stretch + g_domBeg[dir];
182  grid->x[dir][i] = grid->xl[dir][i] + 0.5*m_dx*stretch;
183  grid->xr[dir][i] = grid->xl[dir][i] + m_dx*stretch;
184  grid->dx[dir][i] = m_dx*stretch;
185  #if (CHOMBO_LOGR == YES)
186  }
187  #endif
188  }
189  }
190 
191  CH_assert(m_isBoundarySet);
192  grid->lbound[IDIR] *= left_bound_side[IDIR];
193  grid->rbound[IDIR] *= right_bound_side[IDIR];
194  grid->lbound[JDIR] *= left_bound_side[JDIR];
195  grid->rbound[JDIR] *= right_bound_side[JDIR];
196  grid->lbound[KDIR] *= left_bound_side[KDIR];
197  grid->rbound[KDIR] *= right_bound_side[KDIR];
198  MakeGeometry(grid);
199 
200 // compute cell volumes (dV/m_dx^3) and cylindrical radius
201 
202  #if GEOMETRY != CARTESIAN
203  double mdx_dim = D_EXPAND(m_dx, *m_dx, *m_dx);
204 
205  NX1_TOT = grid->np_tot[IDIR];
206  NX2_TOT = grid->np_tot[JDIR];
207  NX3_TOT = grid->np_tot[KDIR];
208  for (i = 0; i < CHOMBO_NDV; i++) {
209  dV[i] = ArrayMap(NX3_TOT, NX2_TOT, NX1_TOT, a_dV.dataPtr(i));
210  }
211 
212  for (k = 0; k < NX3_TOT; k++) {
213  for (j = 0; j < NX2_TOT; j++) {
214  for (i = 0; i < NX1_TOT; i++) {
215  dV[0][k][j][i] = grid->dV[k][j][i]/mdx_dim;
216  #if CHOMBO_CONS_AM == YES
217  #if GEOMETRY == POLAR || GEOMETRY == CYLINDRICAL
218  dV[1][k][j][i] = grid->x[IDIR][i];
219  #else
220  dV[1][k][j][i] = D_EXPAND(grid->x[IDIR][i], *= sin(grid->x[JDIR][j]);
221  #endif
222  #endif
223  }}}
224 
225 
226  for (i = 0; i < CHOMBO_NDV; i++) FreeArrayMap(dV[i]);
227 #endif /* != CARTESIAN */
228

230 
231 #if (GEOMETRY == CARTESIAN) || (GEOMETRY == CYLINDRICAL)
232 
233  grid->dl_min[IDIR] = m_dx;
234  grid->dl_min[JDIR] = m_dx*g_x2stretch;
235  grid->dl_min[KDIR] = m_dx*g_x3stretch;
236 
237 #else
238 
239  IBEG = grid->lbeg[IDIR]; IEND = grid->lend[IDIR];
240  JBEG = grid->lbeg[JDIR]; JEND = grid->lend[JDIR];
241  KBEG = grid->lbeg[KDIR]; KEND = grid->lend[KDIR];
242 
243  for (idim = 0; idim < 3; idim++) dxmin[idim] = 1.e30;
244 
245  for (i = IBEG; i <= IEND; i++) {
246  for (j = JBEG; j <= JEND; j++) {
247  for (k = KBEG; k <= KEND; k++) {
248 
249  #if (TIME_STEPPING == RK2)
250  D_EXPAND(scrh1 = Length_1(i, j, k, grid); ,
251  scrh2 = Length_2(i, j, k, grid); ,
252  scrh3 = Length_3(i, j, k, grid); )
253 
254  scrh = D_EXPAND(1./scrh1, +1./scrh2, +1./scrh3);
255  scrh = (double)DIMENSIONS/scrh;
256  D_EXPAND(dxmin[IDIR] = MIN (dxmin[IDIR], scrh); ,
257  dxmin[JDIR] = dxmin[IDIR]; ,
258  dxmin[KDIR] = dxmin[IDIR]; )
259  #else
260  scrh = Length_1(i, j, k, grid);
261  dxmin[IDIR] = MIN (dxmin[IDIR], scrh);
262 
263  scrh = Length_2(i, j, k, grid);
264  dxmin[JDIR] = MIN (dxmin[JDIR], scrh);
265 
266  scrh = Length_3(i, j, k, grid);
267  dxmin[KDIR] = MIN (dxmin[KDIR], scrh);
268  #endif
269 
270  }}}
271 
272  grid->dl_min[IDIR] = dxmin[IDIR];
273  grid->dl_min[JDIR] = dxmin[JDIR];
274  grid->dl_min[KDIR] = dxmin[KDIR];

276  #endif
