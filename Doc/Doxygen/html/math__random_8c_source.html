304 double NR_ran2(long int *idum)
329 {
330  int j;
331  long int k;
332  static long idum2=123456789;
333  static long iy=0;
334  static long iv[NTAB];
335  double temp;
336 
337  if (*idum <= 0) { /* Initialize. */
338  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0. */
339  else *idum = -(*idum);
340 
341  idum2=(*idum);
342  for (j = NTAB+7; j >= 0; j--){ /* Load the shuffle table (after 8 warm-ups). */
343 
344  k = (*idum)/IQ1;
345  *idum = IA1*(*idum-k*IQ1)-k*IR1;
346  if (*idum < 0) *idum += IM1;
347  if (j < NTAB) iv[j] = *idum;
348  }
349  iy=iv[0];
350  }
351  k = (*idum)/IQ1; /* Start here when not initializing. */
352  *idum = IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
353  if (*idum < 0) *idum += IM1; /* overflows by Schrage’s method. */
354  k = idum2/IQ2;
355  idum2 = IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise.*/
356 
357  if (idum2 < 0) idum2 += IM2;
358  j = iy/NDIV; /* Will be in the range 0..NTAB-1. */
359  iy = iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
360  iv[j] = *idum; /* are combined to generate output. */
361  if (iy < 1) iy += IMM1;
362  if ((temp=AM*iy) > RNMX) return RNMX; /* Because users don’t expect
363  endpoint values.*/
364  else return temp;
365 }
