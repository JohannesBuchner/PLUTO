770 /* ********************************************************************* */
771 void FARGO_Flux (double *q, double *flx, double eps)
798 {
799  int s;
800  double dqc, d2q, dqp, dqm;
801  static double *dq, *dql, *qp, *qm;
802 
803  if (dq == NULL){
804  dq = ARRAY_1D(NMAX_POINT, double);
805  dql = ARRAY_1D(NMAX_POINT, double);
806  }
807 
808 /* -- compute limited slopes -- */
809 
810  dq[0] = q[1] - q[0];
811  for (s = 1; s < NS_TOT-1; s++){
812  dq[s] = q[s+1] - q[s];
813  if (dq[s]*dq[s-1] > 0.0){
814  dqc = 0.5*(dq[s] + dq[s-1]);
815  d2q = 2.0*(fabs(dq[s]) < fabs(dq[s-1]) ? dq[s]:dq[s-1]);
816  dql[s] = fabs(d2q) < fabs(dqc) ? d2q: dqc;
817  }else dql[s] = 0.0;
818  }
819 
820 /* vanleer_lim(dq,dq-1,dql,JBEG-1,JEND+1,NULL); */
821 
822  #if FARGO_ORDER == 2 /* MUSCL-Hancock, 2nd order */
823  if (eps > 0.0){
824  for (s=SBEG-1; s<=SEND; s++) flx[s] = q[s] + 0.5*dql[s]*(1.0 - eps);
825  }else { /* -- remember: eps > 0 by definition -- */
826  for (s=SBEG-1; s<=SEND; s++) flx[s] = q[s+1] - 0.5*dql[s+1]*(1.0 + eps);
827  }
828  #elif FARGO_ORDER == 3 /* PPM, 3rd order */
829  if (qp == NULL){
830  qp = ARRAY_1D(NMAX_POINT, double);
831  qm = ARRAY_1D(NMAX_POINT, double);
832  }
833  for (s = SBEG-1; s <= SEND+1; s++){
834  dqp = 0.5*dq[s] - (dql[s+1] - dql[s])/6.0;
835  dqm = -0.5*dq[s-1] + (dql[s-1] - dql[s])/6.0;
836  if (dqp*dqm > 0.0) dqp = dqm = 0.0;
837  else{
838  if (fabs(dqp) >= 2.0*fabs(dqm)) dqp = -2.0*dqm;
839  if (fabs(dqm) >= 2.0*fabs(dqp)) dqm = -2.0*dqp;
840  }
841  qp[s] = q[s] + dqp;
842  qm[s] = q[s] + dqm;
843  }
844 
845  if (eps > 0.0){
846  for (s = SBEG-1; s <= SEND; s++){
847  dqc = qp[s] - qm[s];
848  d2q = qp[s] - 2.0*q[s] + qm[s];
849  flx[s] = qp[s] - 0.5*eps*(dqc + d2q*(3.0 - 2.0*eps));
850  }
851  }else{
852  for (s = SBEG-1; s <= SEND; s++){
853  dqc = qp[s+1] - qm[s+1];
854  d2q = qp[s+1] - 2.0*q[s+1] + qm[s+1];
855  flx[s] = qm[s+1] - 0.5*eps*(dqc - d2q*(3.0 + 2.0*eps));
856  }
857  }
858  #else
859  print ("! FARGO_ORDER should be either 2 or 3\n");
860  QUIT_PLUTO(1);
861  #endif
862 
863 }
