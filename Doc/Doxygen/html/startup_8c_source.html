307  #if GEOMETRY == SPHERICAL
308  double thp, thm, sp, sm, dx, dV;

310 
311  if ((dir != IDIR) && (dir != JDIR) && (dir != KDIR)) {
312  print("! Wrong dir in FieldAvergae. Abort! \n");

318 
319  #elif GEOMETRY == CYLINDRICAL
320  if (dir == IDIR) {
321  rp = grid->xr[dir][l];
322  rm = grid->xl[dir][l];
323  bavg = (rp*bp + rm*bm)/(rp + rm);
324  } else bavg = 0.5*(bp + bm);
325 
326  #elif GEOMETRY == POLAR /* Note: Polar is same as cylindrical.
327  Creating separate entries if
328  changes are needed in future. */
329  if (dir == IDIR) {
330  rp = grid->xr[dir][l];
331  rm = grid->xl[dir][l];
332  bavg = (rp*bp + rm*bm)/(rp + rm);
333  } else bavg = 0.5*(bp + bm);
334 
335  #elif GEOMETRY == SPHERICAL
336  if (dir == IDIR) {
337  rp = grid->xr[dir][l];
338  rm = grid->xl[dir][l];
339  dx = grid->dx[dir][l];
340  dV = (rp*rp*rp - rm*rm*rm)/3.;
341  bavg = 0.5*(rp*rp*bp + rm*rm*bm)*dx/dV;
342  }
343  else if (dir == JDIR){
344  thp = grid->xr[dir][l];
345  thm = grid->xl[dir][l];
346  sp = fabs(sin(thp));
347  sm = fabs(sin(thm));
348  dx = grid->dx[dir][l];
349  dV = fabs(cos(thm) - cos(thp));
350  bavg = 0.5*(sp*bp + sm*bm)*dx/dV;
351  } else bavg = 0.5*(bp + bm);
352  #endif
